\begin{frame}{What Inheritance Looks Like in Your Game}
\textbf{Apply what you learned Tuesday:}
\vspace{0.2cm}
\begin{itemize}
  \item Abstract \texttt{Character} --- \texttt{Player} and \texttt{NPC} extend it
  \item Abstract \texttt{Item} --- \texttt{Weapon}, \texttt{Key}, \texttt{Potion} extend it
  \item Abstract \texttt{Question} --- \texttt{MultipleChoice}, \texttt{TrueFalse} extend it
\end{itemize}
\vspace{0.3cm}
\textbf{Every abstract class needs at least one abstract method.}
\begin{itemize}
  \item \texttt{Character}: \texttt{abstract void attack()}
  \item \texttt{Item}: \texttt{abstract void use(Player p)}
  \item \texttt{Question}: \texttt{abstract boolean checkAnswer(String input)}
\end{itemize}
\end{frame}
