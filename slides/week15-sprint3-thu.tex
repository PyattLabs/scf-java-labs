\begin{frame}{Playtest Protocol}
\textbf{Teams swap games.}
\vspace{0.3cm}
\begin{enumerate}
  \item Share code via HedgeDoc or terminal
  \item Other team compiles cold: \texttt{javac *.java}
  \item Other team runs: \texttt{java Main}
  \item Play for 10 minutes without asking for help
  \item Write feedback --- 3 to 5 sentences minimum
\end{enumerate}
\vspace{0.3cm}
\textit{If they cannot compile without your help, that is feedback.}
\end{frame}
\begin{frame}{Feedback Requirements}
\textbf{Written feedback submitted before end of class.}
\vspace{0.3cm}
\begin{itemize}
  \item What worked well
  \item What was confusing or unclear
  \item One specific suggestion for improvement
  \item Whether the game compiled and ran without help
\end{itemize}
\vspace{0.3cm}
\textit{Bad: "It was good."}\\
\textit{Good: "Input prompt was unclear --- I didn't know valid commands."}
\end{frame}
